              <span className="absolute left-2 text-stone-200 text-sm font-bold opacity-70 z-0">玩家跑道</span>

              <div

                className={`absolute text-5xl sm:text-6xl transition-all duration-700 ease-out z-20 drop-shadow-md ${gameState === 'playing' ? 'anim-gallop' : 'facing-right'}`}

                style={{ left: `calc(${playerPos}% - 10px)` }}

              >

                🏇

              </div>

            </div>

            {/* 跑道 2：電腦 1 */}

            <div className="relative h-20 sm:h-24 border-b-4 border-dashed border-white/40 bg-green-700/50 flex items-center px-2">

              <span className="absolute left-2 text-stone-200 text-sm font-bold opacity-70 z-0">電腦 1號</span>

              <div

                className={`absolute text-5xl sm:text-6xl transition-all duration-1000 ease-linear z-20 drop-shadow-md ${gameState === 'playing' ? 'anim-gallop-slow' : 'facing-right'}`}

                style={{ left: `calc(${bot1Pos}% - 10px)` }}

              >

                🐎

              </div>

            </div>

            {/* 跑道 3：電腦 2 */}

            <div className="relative h-20 sm:h-24 bg-green-800/50 flex items-center px-2">

              <span className="absolute left-2 text-stone-200 text-sm font-bold opacity-70 z-0">電腦 2號</span>

              <div

                className={`absolute text-5xl sm:text-6xl transition-all duration-1000 ease-linear z-20 drop-shadow-md ${gameState === 'playing' ? 'anim-gallop-fast' : 'facing-right'}`}

                style={{ left: `calc(${bot2Pos}% - 10px)` }}

              >

                🎠

              </div>

            </div>

          </div>

          {/* 答題區 */}

          <div className="w-full max-w-4xl bg-white/95 rounded-2xl shadow-xl flex flex-col overflow-hidden relative border-4 border-stone-300 z-10">

           

            {/* 計時條 */}

            <div className="w-full h-3 bg-stone-200">

              <div

                className={`h-full transition-all duration-1000 linear ${timeLeft > 5 ? 'bg-blue-500' : 'bg-red-500'}`}

                style={{ width: `${(timeLeft / 15) * 100}%` }}

              ></div>

            </div>

            {/* 題目區域 */}

            <div className="p-6 sm:p-8 flex-1">

              <div className="flex justify-between items-start mb-6">

                <h3 className="text-xl sm:text-3xl font-bold text-stone-800 leading-snug">

                  Q: {currentQ?.q}

                </h3>

                <div className={`text-2xl font-black rounded-full w-12 h-12 flex items-center justify-center shrink-0 ml-4 shadow-inner ${timeLeft > 5 ? 'bg-stone-100 text-stone-600' : 'bg-red-100 text-red-600 animate-pulse'}`}>

                  {timeLeft}

                </div>

              </div>

              {/* 選項按鈕 */}

              <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 mt-4">

                {currentQ?.options.map((option, idx) => {

                  let btnClass = "bg-stone-100 hover:bg-stone-200 border-stone-300 text-stone-700";

                 

                  if (feedback) {

                    if (option === currentQ.answer) {

                      btnClass = "bg-green-500 border-green-600 text-white shadow-lg transform scale-105";

                    } else if (feedback === 'wrong' || feedback === 'timeout') {

                      btnClass = "bg-stone-200 border-stone-300 text-stone-400 opacity-50";

                    }

                  }

                  return (

                    <button

                      key={idx}

                      disabled={feedback !== null}

                      onClick={() => handleAnswer(option)}

                      className={`border-2 rounded-xl p-4 text-lg sm:text-xl font-bold transition-all text-left ${btnClass} ${feedback ? 'cursor-not-allowed' : 'active:scale-95 hover:shadow-md'}`}

                    >

                      {String.fromCharCode(65 + idx)}. {option}

                    </button>

                  );

                })}

              </div>

            </div>

            {/* 回饋疊加層 (答對/答錯提示) */}

            {feedback && (

              <div className="absolute inset-0 flex items-center justify-center bg-white/60 backdrop-blur-sm z-30 animate-in fade-in duration-200">

                {feedback === 'correct' && (

                  <div className="bg-green-500 text-white px-8 py-6 rounded-2xl shadow-2xl transform scale-110 flex flex-col items-center">

                    <span className="text-6xl mb-2">✅</span>

                    <span className="text-3xl font-black">答對了！馬匹加速！</span>

                  </div>

                )}

                {feedback === 'wrong' && (

                  <div className="bg-red-500 text-white px-8 py-6 rounded-2xl shadow-2xl transform scale-110 flex flex-col items-center">

                    <span className="text-6xl mb-2">❌</span>

                    <span className="text-3xl font-black">答錯了！</span>

                  </div>

                )}

                {feedback === 'timeout' && (

                  <div className="bg-orange-500 text-white px-8 py-6 rounded-2xl shadow-2xl transform scale-110 flex flex-col items-center">

                    <span className="text-6xl mb-2">⏱️</span>

                    <span className="text-3xl font-black">時間到！</span>

                  </div>

                )}

              </div>

            )}

          </div>

        </div>

      );

    }

    // 將 React 應用渲染到畫面上

    const root = ReactDOM.createRoot(document.getElementById('root'));

    root.render(<App />);
